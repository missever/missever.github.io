#include<bits/stdc++.h>
#define MAX 1000000007
using namespace std;
map<int,int>p;
vector<int>f;
int a[1000005],b;
int d[1000005];

void solve()
{
    int i,l,r,mid,ans = 0;
    for(i = 0;i < f.size(); i++)
    {
        l = 1;
        r = ans;
        while(l <= r)
        {
            mid = (l + r) >> 1;
            if(d[mid] < f[i]) l = mid + 1;
            else r = mid - 1;
        }
        d[l] = f[i];
        if(l > ans) ans = l;
    }
    printf("%d\n",ans +1);
}

int main(){
    int n,m,i;
    scanf("%d%d",&n,&m);
    for(i = 1;i <= n; i++)
    {
        scanf("%d",&a[i]);
        p[a[i]] = i;
    }
    for(i = 1;i <= m; i++)
    {
        scanf("%d",&b);
        if(p[b]) f.push_back(p[b]);
    }
    solve();
    return 0;
}
