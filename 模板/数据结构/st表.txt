const int maxn = 1e5 + 5;
int a[maxn];
int st[maxn][22];
int lg[maxn];

void st_init(int len) {
    for (int i = len - 1; i >= 0; --i) {
        st[i][0] = a[i + 1];
        for (int j = 1; i + (1 << j) <= len; ++j)
            st[i][j] = max(st[i][j - 1], st[i + (1 << (j - 1))][j - 1]);
    }
}
int Max(int l, int r) {
    int k = lg[r - l + 1];
    return max(st[l][k], st[r - (1 << k) + 1][k]);
}


for (int i = 2; i < maxn; ++i) lg[i] = lg[i >> 1] + 1;
